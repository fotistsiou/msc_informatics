- Connect as postgres user:
  fotistsiou@fotistsiou:~$ sudo -i -u postgres 
  
- Connect with psql tool:
  postgres@fotistsiou:~$ psql 

- Show the databases:
  postgres=# \l  

- Show the users:
  postgres=# \du

- Clear console:
  postgres=# \! clear  

- Exit from psql tool:
  postgres=# \q 

- Exit from postgreSQL:
  exit

- Create role:
  postgres=# CREATE ROLE datamaster createdb login password '2410'; 

- Create database:
  postgres=# CREATE DATABASE myunipi OWNER datamaster;   

- Login database:
  postgres=# \c myunipi 

- Open editor:
  myunipi=# \e

- Open student table schema
  myunipi=# \d student 

- Create department and student tables
  myunipi=# create table department (
                dept_name varchar(20),
                building varchar(15),
                budget numeric(12,2) check (budget > 0),
                primary key (dept_name) 
            );

  myunipi=# create table student (
                ID varchar(5),
                name varchar(20) not null,
                dept_name varchar(20),
                tot_cred numeric(3,0) check (tot_cred >= 0),
                primary key (ID),
                foreign key (dept_name) references department (dept_name)
                on delete set null
            );
